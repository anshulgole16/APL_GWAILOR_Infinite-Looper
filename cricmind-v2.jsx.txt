const GEMINI_KEY = ""; // Set your Gemini API key here for hackathon demo
const GEMINI_URL = `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${GEMINI_KEY}`;

async function callGemini(systemPrompt, userPrompt, imageBase64 = null) {
  // Build parts
  const parts = [];
  if (imageBase64) {
    parts.push({ inlineData: { mimeType: "image/jpeg", data: imageBase64 } });
  }
  parts.push({ text: userPrompt });

  const body = {
    systemInstruction: { parts: [{ text: systemPrompt }] },
    contents: [{ role: "user", parts }],
    generationConfig: { maxOutputTokens: 1024, temperature: 0.8 },
  };

  const res = await fetch(GEMINI_URL, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(body),
  });
  const data = await res.json();
  if (data.error) throw new Error(data.error.message);
  return data.candidates?.[0]?.content?.parts?.[0]?.text || "No response.";
}

// ─── Typing animation hook ─────────────────────────────────────────────────
function useTypingEffect(text, speed = 12) {
  const [displayed, setDisplayed] = useState("");
  const [done, setDone] = useState(false);
  useEffect(() => {
    setDisplayed("");
    setDone(false);
    if (!text) return;
    let i = 0;
    const id = setInterval(() => {
      i++;
      setDisplayed(text.slice(0, i));
      if (i >= text.length) { clearInterval(id); setDone(true); }
    }, speed);
    return () => clearInterval(id);
  }, [text]);
  return { displayed, done };
}

// ─── Voice hook ────────────────────────────────────────────────────────────
function useVoice(onResult) {
  const recRef = useRef(null);
  const [listening, setListening] = useState(false);
  const start = () => {
    const SR = window.SpeechRecognition || window.webkitSpeechRecognition;
    if (!SR) return alert("Voice not supported. Use Chrome/Edge.");
    const r = new SR(); r.lang = "en-IN"; r.interimResults = false;
    r.onresult = e => onResult(e.results[0][0].transcript);
    r.onend = () => setListening(false);
    r.onerror = () => setListening(false);
    recRef.current = r; r.start(); setListening(true);
  };
  const stop = () => { recRef.current?.stop(); setListening(false); };
  return { listening, start, stop };
}

// ─── Markdown renderer ─────────────────────────────────────────────────────
function MD({ text }) {
  return (
    <div className="space-y-1 text-sm leading-relaxed">
      {text.split("\n").map((line, i) => {
        if (!line.trim()) return <div key={i} className="h-1.5" />;
        if (line.startsWith("## ")) return <h2 key={i} className="font-black text-amber-400 text-base mt-3 mb-1">{line.slice(3)}</h2>;
        if (line.startsWith("### ")) return <h3 key={i} className="font-bold text-amber-300 mt-2">{line.slice(4)}</h3>;
        if (line.startsWith("- ") || line.startsWith("* ")) {
          const html = line.slice(2).replace(/\*\*(.*?)\*\*/g, "<b class='text-amber-200'>$1</b>");
          return <div key={i} className="flex gap-2 items-start"><span className="text-amber-500 mt-0.5 flex-shrink-0">▸</span><span dangerouslySetInnerHTML={{ __html: html }} /></div>;
        }
        if (/^\d+\.\s/.test(line)) {
          const m = line.match(/^(\d+)\.\s(.+)/);
          if (m) {
            const html = m[2].replace(/\*\*(.*?)\*\*/g, "<b class='text-amber-200'>$1</b>");
            return <div key={i} className="flex gap-2"><span className="text-amber-500 font-bold min-w-[1.4rem]">{m[1]}.</span><span dangerouslySetInnerHTML={{ __html: html }} /></div>;
          }
        }
        const html = line.replace(/\*\*(.*?)\*\*/g, "<b class='text-amber-200'>$1</b>").replace(/\*(.*?)\*/g, "<i class='text-green-300'>$1</i>");
        return <p key={i} dangerouslySetInnerHTML={{ __html: html }} />;
      })}
    </div>
  );
}

// ─── Radar Chart ──────────────────────────────────────────────────────────
function RadarChart({ data }) {
  const size = 100;
  const cx = 50, cy = 50, r = 35;
  const keys = Object.keys(data);
  const vals = Object.values(data);
  const n = keys.length;

  const getPoint = (i, val) => {
    const angle = (2 * Math.PI * i) / n - Math.PI / 2;
    const dist = (val / 100) * r;
    return { x: cx + dist * Math.cos(angle), y: cy + dist * Math.sin(angle) };
  };

  const gridLevels = [25, 50, 75, 100];
  const polyPoints = vals.map((v, i) => { const p = getPoint(i, v); return `${p.x},${p.y}`; }).join(" ");

  return (
    <svg viewBox="0 0 100 100" className="w-full max-h-36">
      {/* Grid */}
      {gridLevels.map(lv =>
        <polygon key={lv} points={keys.map((_, i) => { const p = getPoint(i, lv); return `${p.x},${p.y}`; }).join(" ")}
          fill="none" stroke="#ffffff10" strokeWidth="0.5" />
      )}
      {/* Axes */}
      {keys.map((_, i) => { const p = getPoint(i, 100); return <line key={i} x1={cx} y1={cy} x2={p.x} y2={p.y} stroke="#ffffff15" strokeWidth="0.5" />; })}
      {/* Data */}
      <polygon points={polyPoints} fill="#f59e0b30" stroke="#f59e0b" strokeWidth="1.5" />
      {vals.map((v, i) => { const p = getPoint(i, v); return <circle key={i} cx={p.x} cy={p.y} r="2" fill="#f59e0b" />; })}
      {/* Labels */}
      {keys.map((k, i) => {
        const p = getPoint(i, 115);
        return <text key={i} x={p.x} y={p.y} textAnchor="middle" fontSize="5.5" fill="#9ca3af" dominantBaseline="middle">{k}</text>;
      })}
    </svg>
  );
}

// ─── STRATEGY PAGE ─────────────────────────────────────────────────────────
function StrategyPage() {
  const [form, setForm] = useState({ role: "batting", opponent: "", pitch: "", overs: "", wickets: "" });
  const [result, setResult] = useState(null);
  const [loading, setLoading] = useState(false);
  const [fieldType, setFieldType] = useState("defensive");
  const [aggression, setAggression] = useState(50);
  const [winProb, setWinProb] = useState({ batting: 50, bowling: 50 });

  const pitchTypes = ["Flat", "Seaming", "Spinning", "Bouncy", "Slow & Low"];
  const opponents = ["Spin-heavy", "Pace attack", "All-rounders", "Balanced", "Inexperienced"];

  const analyze = async () => {
    if (loading) return;
    setLoading(true); setResult(null);

    const overs = parseInt(form.overs) || 20;
    const wickets = parseInt(form.wickets) || 0;
    const aggrValue = form.role === "batting"
      ? Math.min(95, 40 + (20 - overs) * 2 + wickets * 3)
      : Math.max(20, 70 - wickets * 5);
    setAggression(aggrValue);
    setFieldType(aggrValue > 60 ? "aggressive" : "defensive");

    const battingWin = form.role === "batting"
      ? Math.min(85, Math.max(20, 60 - wickets * 5 + (overs > 10 ? 10 : -10)))
      : Math.min(80, Math.max(20, 40 + wickets * 8));
    setWinProb({ batting: battingWin, bowling: 100 - battingWin });

    const SYSTEM = `You are CricMind AI — elite cricket tactician. Give match strategy in exactly this format:

## 🎯 Situation Read
2 lines assessing the match situation critically.

## 📋 Primary Strategy
The main approach right now (3-4 lines).

## ⚡ Key Tactics
- Tactic 1 (specific)
- Tactic 2 (specific)
- Tactic 3 (specific)
- Tactic 4 (specific)

## 🎲 If Plan A Fails
Quick contingency (2 lines).

## 🏆 Win Condition
One sentence on the path to victory.

Be direct, tactical, use real cricket terms.`;

    try {
      const reply = await callGemini(SYSTEM,
        `Situation: ${form.role === "batting" ? "WE BATTING" : "WE BOWLING"}. Opponent: ${form.opponent || "unknown"}. Pitch: ${form.pitch || "normal"}. Overs left: ${form.overs || "?"}. Wickets ${form.role === "batting" ? "lost" : "taken"}: ${form.wickets || "?"}.`
      );
      setResult(reply);
    } catch (e) {
      setResult(`❌ ${e.message}`);
    }
    setLoading(false);
  };

  return (
    <div className="overflow-y-auto h-full">
      <div className="px-4 py-4 space-y-3">
        {/* Role Toggle */}
        <div className="flex p-1 bg-white/5 border border-white/10 rounded-2xl">
          {[["batting", "🏏 Batting"], ["bowling", "🎯 Bowling"]].map(([v, l]) => (
            <button key={v} onClick={() => setForm(f => ({ ...f, role: v }))}
              className={`flex-1 py-2.5 rounded-xl font-bold text-sm transition-all ${form.role === v ? "bg-amber-500 text-black" : "text-gray-400"}`}>{l}</button>
          ))}
        </div>

        {/* Filters */}
        <div>
          <p className="text-[11px] text-gray-500 font-semibold mb-1.5">Opponent Style</p>
          <div className="flex flex-wrap gap-1.5">
            {opponents.map(o => (
              <button key={o} onClick={() => setForm(f => ({ ...f, opponent: o }))}
                className={`text-xs px-2.5 py-1.5 rounded-xl border transition-all ${form.opponent === o ? "bg-green-500/20 border-green-500/40 text-green-300" : "bg-white/5 border-white/10 text-gray-400"}`}>{o}</button>
            ))}
          </div>
        </div>
        <div>
          <p className="text-[11px] text-gray-500 font-semibold mb-1.5">Pitch Condition</p>
          <div className="flex flex-wrap gap-1.5">
            {pitchTypes.map(p => (
              <button key={p} onClick={() => setForm(f => ({ ...f, pitch: p }))}
                className={`text-xs px-2.5 py-1.5 rounded-xl border transition-all ${form.pitch === p ? "bg-blue-500/20 border-blue-500/40 text-blue-300" : "bg-white/5 border-white/10 text-gray-400"}`}>{p}</button>
            ))}
          </div>
        </div>

// ─── SCREENSHOT ANALYZER ───────────────────────────────────────────────────
function ScannerPage() {

    if (!file) return;
    const reader = new FileReader();
    reader.onload = (e) => {
      setImage(e.target.result);
      setImageB64(e.target.result.split(",")[1]);
      setResult(null);
    };
    reader.readAsDataURL(file);
  };

  const analysisTypes = [
    { id: "strategy", label: "Strategy", emoji: "📊" },
    { id: "scorecard", label: "Scorecard", emoji: "📋" },
    { id: "fieldset", label: "Field Setup", emoji: "🏟️" },
    { id: "wagon", label: "Wagon Wheel", emoji: "🎯" },
  ];

  const SYSTEMS = {
    strategy: `You are CricMind AI Vision. Analyze this cricket screenshot and provide:
## 🔍 Match Situation
What you see in the image.
## ⚡ Immediate Strategy
Best tactical move right now.
## 📋 Key Observations
3-4 bullet points from the image.
## 🏆 Recommendation
One clear action to take.`,
    scorecard: `Analyze this cricket scorecard/screenshot. Identify key stats, batting/bowling performances, partnerships, and overall match situation. Give insights on who's winning and why.`,
    fieldset: `Analyze this cricket field placement. Identify fielder positions, gaps in the field, which shots are open, and what bowling strategy is being used. Suggest how a batsman should exploit this field.`,
    wagon: `Analyze this wagon wheel or shot chart. Identify the batsman's scoring zones, weaknesses, preferred shots, and suggest how a bowler should bowl to exploit the gaps.`,
  };

  const analyze = async () => {
    if (!imageB64 || loading) return;
    setLoading(true); setResult(null);
    try {
      const reply = await callGemini(SYSTEMS[analysisType], "Analyze this cricket image thoroughly.", imageB64);
      setResult(reply);
    } catch (e) {
      setResult(`❌ ${e.message}\n\n*Gemini Vision requires a valid API key with vision access.*`);
    }
    setLoading(false);
  };

  return (
    <div className="overflow-y-auto h-full">
      <div className="px-4 py-4 space-y-4">
        {/* Header */}
        <div className="bg-gradient-to-br from-blue-900/40 to-blue-950/60 border border-blue-500/20 rounded-2xl p-4">
          <div className="flex items-center gap-2 mb-1">
            <span className="text-2xl">📸</span>

              <h2 className="text-white font-black">Screenshot Analyzer</h2>

            </div>
          </div>

        </div>

        {/* Analysis Type */}
        <div className="grid grid-cols-4 gap-2">
          {analysisTypes.map(t => (
            <button key={t.id} onClick={() => setAnalysisType(t.id)}
              className={`rounded-xl p-2.5 text-center transition-all border ${analysisType === t.id ? "bg-blue-500/20 border-blue-500/40" : "bg-white/5 border-white/10"}`}>

        >

          </div>
        )}
      </div>
    </div>
  );
}

  coach: "AI Coach",
  strategy: "Match Strategy",
  commentary: "AI Commentary",
  scanner: "Screenshot Analyzer",
  dashboard: "Dashboard",
};

// ─── APP ───────────────────────────────────────────────────────────────────
export default function App() {
  const [page, setPage] = useState("home");
  const [coachMode, setCoachMode] = useState("batter");

  return (
    <div className="flex flex-col h-screen max-w-md mx-auto select-none"
      style={{
        background: "linear-gradient(160deg, #050a05 0%, #0a0f0a 40%, #050908 100%)",
        fontFamily: "'Segoe UI Variable', 'Segoe UI', system-ui, sans-serif",
      }}>

      {/* Header */}
      <div className="flex items-center justify-between px-4 py-3 border-b border-white/5 flex-shrink-0 bg-black/20 backdrop-blur-xl">
        <div className="flex items-center gap-2">
          <span className="text-xl">🏏</span>
          <span className="text-white font-black text-base tracking-tight">{PAGE_TITLES[page]}</span>
        </div>
        <div className="flex items-center gap-2">
          <span className="text-[10px] text-amber-400/70 font-semibold tracking-wider">GEMINI</span>
          <div className="w-2 h-2 rounded-full bg-green-500 animate-pulse" />
        </div>
      </div>

      {/* Live Ticker */}
      <Ticker />

      {/* Content */}
      <div className="flex-1 overflow-hidden">
        {page === "home" && <HomePage setPage={setPage} coachMode={coachMode} setCoachMode={setCoachMode} />}
        {page === "coach" && <CoachPage coachMode={coachMode} />}
        {page === "strategy" && <StrategyPage />}
        {page === "commentary" && <CommentaryPage />}
        {page === "scanner" && <ScannerPage />}
        {page === "dashboard" && <DashboardPage setPage={setPage} coachMode={coachMode} />}
      </div>

      {/* Bottom Nav */}
      <div className="border-t border-white/5 bg-black/40 backdrop-blur-xl flex-shrink-0">
        <div className="flex">
          {NAV.map(n => (
            <button key={n.id} onClick={() => setPage(n.id)}
              className={`flex-1 flex flex-col items-center py-2.5 gap-0.5 transition-all relative ${page === n.id ? "text-amber-400" : "text-gray-600 hover:text-gray-400"}`}>
              {page === n.id && <div className="absolute top-0 left-1/2 -translate-x-1/2 w-8 h-0.5 bg-amber-400 rounded-full" />}
              <span className={`text-base transition-transform ${page === n.id ? "scale-110" : ""}`}>{n.emoji}</span>
              <span className="text-[9px] font-bold">{n.label}</span>
            </button>
          ))}
        </div>
      </div>
    </div>
  );
}
